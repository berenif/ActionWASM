Rust Game Logic

// Combat system with hitbox detection
fn handle_combat(
    mut commands: Commands,
    time: Res<Time>,
    mut query: Query<(&mut Health, &Transform, &Hitbox)>,
) {
    for (mut health, transform, hitbox) in query.iter_mut() {
        // Check collisions and apply damage
        if let Some(damage) = check_hitbox_collision(transform, hitbox) {
            health.current -= damage;
            spawn_damage_number(&mut commands, transform.translation, damage);
        }
    }
}

WebRTC Networking

// P2P connection setup
async function createPeerConnection(roomCode: string) {
    const peer = new SimplePeer({
        initiator: true,
        trickle: false,
        config: {
            iceServers: [
                { urls: 'stun:stun.l.google.com:19302' }
            ]
        }
    });
    
    peer.on('signal', data => {
        sendSignal(roomCode, data);
    });
    
    return peer;
}
